      author         = {Martin, Joel, T. and Spitschan, Manuel and Baker, Daniel, H. and Wade, Alex},

      year           = {2022},
      publisher      = {Zenodo},
      version        = {v0.0.1},
      doi            = {10.5281/zenodo.1234},
      url            = {https://github.com/PySilentSubstitution/silentsub}
    }
